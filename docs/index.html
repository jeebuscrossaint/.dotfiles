install

Linux

sh 
# clone and stow
$ git clone https://github.com/jeebuscrossaint/.dotfiles ~/.dotfiles
$ cd ~/.dotfiles && stow linux
$ coat apply

# canvas module (optional)
$ echo "YOUR_URL" > ~/.config/canvas/ical-url

# nerd fonts on openbsd
$ ./install-nerdfonts.sh

requires gnu stow and coat

Windows

powershell 
# install scoop
PS> irm get.scoop.sh | iex
PS> scoop bucket add extras

# install packages
PS> scoop install glazewm pwsh lf btop `
      uutils-coreutils helix fastfetch

# install dotfiles
PS> git clone https://github.com/jeebuscrossaint/.dotfiles
PS> cd .dotfiles && .\install.ps1
